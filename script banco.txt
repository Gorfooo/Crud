create database site
use site
create table login(
    email varchar(50),
    senha varchar(50),
    nome varchar(50),
    nascimento date,
    empresa varchar(50)
    )
create table tb_cliente(
    id_cliente int not null AUTO_INCREMENT primary key,
    nome varchar(50),
    cep varchar(9),
    numero int,
    logradouro varchar(50),
    cidade varchar(50),
    UF varchar(50),
    pais varchar(56),
    limite_credito float,
    cpf varchar(14),
    cnpj varchar(18)
    )
create table tb_produto(
    id_produto int not null AUTO_INCREMENT primary key,
    descricao varchar(50),
    preco float,
    id_unidade_medida int not null,
    custo float,
    quantidade float,
    status char(1)
    )
create table tb_unidade_medida(
    id_unidade_medida int not null AUTO_INCREMENT primary key,
    descricao varchar(50)
    )
alter table tb_produto add constraint fk_unidade_medida foreign key (id_unidade_medida) references tb_unidade_medida (id_unidade_medida)
create table tb_venda(
    id_venda int not null AUTO_INCREMENT primary key,
    data date,
    id_cliente int not null
    )
alter table tb_venda add constraint fk_venda_cliente foreign key (id_cliente) references tb_cliente (id_cliente)
create table tb_produto_venda(
    id_produto int not null,
    id_venda int not null,
    valor_unit float,
    quantidade float
    )
alter table tb_produto_venda add constraint pk_produto_venda primary key (id_produto, id_venda)
alter table tb_produto_venda add constraint fk_produto foreign key (id_produto) references tb_produto (id_produto)
alter table tb_produto_venda add constraint fk_venda foreign key (id_venda) references tb_venda (id_venda)
